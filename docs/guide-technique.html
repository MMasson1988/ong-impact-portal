#!/usr/bin/env python
import os
import sys

def main():
    os.environ.setdefault(
        "DJANGO_SETTINGS_MODULE",
        "config.settings.development",
    )
    from django.core.management import execute_from_command_line
    execute_from_command_line(sys.argv)

if __name__ == "__main__":
    main()

5. Applications & modèles principaux

Programme (extrait)

class Program(models.Model):
    name = models.CharField(max_length=200)
    slug = models.SlugField(unique=True)
    summary = models.TextField()
    is_published = models.BooleanField(default=True)
    is_featured = models.BooleanField(default=False)

Document avec fichier privé

class Document(models.Model):
    title = models.CharField(max_length=200)
    file = models.FileField(upload_to="documents/", blank=True, null=True)
    private_file = models.FileField(upload_to="private/", blank=True, null=True)
    is_private = models.BooleanField(default=False)

6. Authentification & rôles

# config/settings/base.py
AUTH_USER_MODEL = "accounts.User"
LOGIN_URL = "accounts:login"
LOGIN_REDIRECT_URL = "dashboards:staff_home"

# Mixin contrôle d'accès staff
class StaffRequiredMixin(LoginRequiredMixin, UserPassesTestMixin):
    def test_func(self):
        user = self.request.user
        return user.is_authenticated and (
            user.is_superuser
            or user.groups.filter(
                name__in=["SuperAdmin", "Communicator", "ProgramManager", "Staff"]
            ).exists()
        )
# Management command
python manage.py setup_roles

7. Protection des médias privés

# Ne PAS monter private_media/ en URL statique publique.

class PrivateFileDownloadView(StaffRequiredMixin, View):
    def get(self, request, pk):
        document = get_object_or_404(
            Document, pk=pk, is_private=True, is_published=True
        )
        return FileResponse(
            document.private_file.open("rb"),
            as_attachment=True,
            filename=document.private_file.name.split("/")[-1],
        )
